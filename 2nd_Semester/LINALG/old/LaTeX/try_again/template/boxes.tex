%to make boxes:
%tcbset is used to make different options for tcboxes and tcolourboxes which can be added as argument, it basically defines the standard formatting for all following boxes
%using these keywords as arguments for the boxes you can also adjust them more
\tcbset{
	before=\par\noindent, 
	after=\par,
	basicbox/.style = {
		enhanced,
		top=0.5mm,
		bottom=0.5mm,
		left=1mm,
		right=1mm,
		colback=white,
		boxrule=0pt,
		leftrule=1mm,
		arc=1mm,
        sharp corners = all,
	},
    important/.style={
		leftrule=1mm,
		rightrule=1mm,
		colframe = important,
	},
     noframe/.style={
        basicbox,
        colframe = white,
    },
    rounded/.style={
        basicbox,
        rounded corners = all,
    },
	headerdef/.style={
		fonttitle=\sffamily\bfseries\boldmath,
		detach title,
		before upper={\tcbtitle\quad},
	}
}
