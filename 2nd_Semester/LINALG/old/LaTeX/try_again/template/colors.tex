%pagestyles:

%titlepage
\newcommand{\createtitlepagestyle}{
    \fancypagestyle{TitlePageStyle}{
%        \thispagestyle{empty}
    }
}

%mainpage
\newcommand{\createmainpagestyle}{
    \fancypagestyle{main}{
        \fancyhf{}
        % \rfoot{Seite \thepage \protect{ }von \pageref{LastPage}}
        \renewcommand{\headrulewidth}{0pt}
        \renewcommand{\footrulewidth}{0pt}
    }
}

%plain page with page number
\newcommand{\overrideplainstyle}{
    \fancypagestyle{plain}{
        \fancyhf{}
        \rfoot{Seite \thepage \protect{ }von \pageref{LastPage}}
        \renewcommand{\headrulewidth}{0pt}
        \renewcommand{\footrulewidth}{0pt}
    }
}

\renewcommand{\maketitle}{
    \begin{center}
        \sffamily
        \begin{tabular}{l C{35mm} r}
            \multirow{2}{*}{\LARGE \textbf{\themaintitle}} & \theauthor\\& \thedate
        \end{tabular}
    \end{center}
}
